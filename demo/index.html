Javascript

$( document ).ready(function() {
    var feather = $('#mycarousel').feather({
        loop : false,
        currentItem_class : 'current',
    }).data('api_feather');
    $('#mycarousel_nav_next').click(function()
    {  
        feather.nextSlide(); 
    });
    $('#mycarousel_nav_prev').click(function()
    {
        feather.prevSlide();
    });
});
